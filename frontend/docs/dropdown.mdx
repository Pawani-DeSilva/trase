## Dropdown

<div className="components-container -spaced -width-200">
  <State initial={{ options:
    [{ label: 'Blue', value: 'blue' },
      { label: 'Red', value: 'red' },
      { label: 'Green', value: 'green' }],
      value: { label: 'Red', value: 'red' }
    }}>
    {({ state, setState }) => (
      <Dropdown
        options={state.options}
        value={state.value}
        onChange={value => setState({ value })}
      />
    )}
  </State>
</div>
